GIT commands

//setup
git init

//add file
git add
//add all in current folder
git add .
//add everything everywhere
git add -A

//git status of repo
git status

//commit files
git commit -m “message”
//see commit log
git log

//list branches
git branch
//checkout branch
git checkout
//checkout master
git checkout master
//checkout branch
git checkout 7digitcommitnumber

git merge BRANCHNAME

//git remote operation
git remote add origin SSHURL
git pull origin master
git push origin master
